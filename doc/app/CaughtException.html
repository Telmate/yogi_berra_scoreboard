def self.search(query)
  if query.blank?
    scoped
  else
    sql = query.split.map do |word|
      %w[first_name last_name].map do |column|
      sanitize_sql ["#{column} LIKE ?", "%#{word}%"]
      end.join(" or ")
    end.join(") and (")
    where("(#{sql})")
  end
end
